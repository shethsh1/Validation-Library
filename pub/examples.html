custom function

back to top


Code for the options of above 

            
 <form id="customFunction">
    <h2>custom function</h2>
    <a class="back" href="#content">back to top</a> <br />
    <label for="firstname">
    First Name (Try starting the name with an s)
    </label
    > <br />
    <input type="text" name="firstname" /> <br />
 </form>
            
            
 const customFunction = new Validation("#customFunction")
 customFunction.validate({
    firstname: {
        type: "text",
        required: true,
        customCheck: {
            checker: function (val) {
                if (val) {
                    if (val[0] === "S" || val[0] === "s") {
                        return true
                    }
                }
                return false
            },
            errorMessage: "No names that start with s"
        }
    },
})

.getActiveErrors() example

back to top








Code for the options of above

.getActiveErrors returns all the active errors that are on this form, for example the above form will return

const commentForm3 = new Validation("#getActiveErrorsExample")
commentForm3.validate({
    firstname: {
        type: "text",
        minLength: 1,
        numbers: false,
    },
    email: {
        type: "email"
    },
    phonenumber: {
        type: "phoneNumber"
    },
    comment: {
        type: "textarea",
        minLength: 2,
        maxLength: 25,
    }
})
// get active errors will return format is not valid for email 
// and length is less than 2 for comment. E.g. in console
> commentForm3.getActiveErrors()
{
    comment: ["Length must be at least 2"],
    email: ["Format is not valid"]
}

.editOptions(name, newOptions) example

back to top








Code for the options of above

.editOptions(name, newOptions) will edit an existing form given name and new options for that name

const commentForm4 = new Validation("#editOptionsExample")
commentForm4.validate({
    firstname: {
        type: "text",
        minLength: 1,
        numbers: false,
    },
    email: {
        type: "email"
    },
    phonenumber: {
        type: "phoneNumber"
    },
    comment: {
        type: "textarea",
        minLength: 2,
        maxLength: 25,
    }
})
// above will edit the firstname field
commentForm4.editOptions("firstname", { 
    type: "text", 
    minLength: 1000 
})
